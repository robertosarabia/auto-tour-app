package com.example.android.miwok; /** * {@Link Location} represents a location that the user wants to visit. * It contains a name, address, phone number, and image for each location. */ public class Location { /** * Name of the location */ private String mLocationName; /** * Address of location */ private String mLocationAddress; /** * Phone number of location */ private String mLocationPhone; /** * drawable resource ID associated with location */ private int mImageResourceId; /** * Create a Location object with image * * @param locationName is the name of the location * @param locationAddress is the address of the location * @param locationPhone is the phone number of the location * @param imageResourceId is the drawable resource ID associated with the location */ public Location(String locationName, String locationAddress, String locationPhone, int imageResourceId) { mLocationName = locationName; mLocationAddress = locationAddress; mLocationPhone = locationPhone; mImageResourceId = imageResourceId; } /** * Get the name of the location */ public String getLocationName() { return mLocationName; } /** * Get the address of the location */ public String getLocationAddress() { return mLocationAddress; } /** * Get the phone number of the location */ public String getLocationPhone() { return mLocationPhone; } /** * Get the image resource ID of the location */ public int getImageResourceId() { return mImageResourceId; } }
